    <script>
        function calculateScore() {
            let form = document.getElementById('transportForm');
            let totalCost = 0, totalTime = 0;
            for (let i = 0; i < form.length; i++) {
                if (form[i].checked) {
                    let values = form[i].value.split(',');
                    totalCost += parseInt(values[0]);
                    totalTime += parseInt(values[1]);
                }
            }
            let totalScore = (totalCost * 0.9) + (totalTime * 0.1);
            document.getElementById('result').innerHTML = `Total Cost: ${totalCost}, Total Time: ${totalTime} days, Total Score: ${totalScore.toFixed(2)}`;
        }
    }
</script>
